<script>

let pontuacao = 0
let respostasAluno = []

function corrigir(){

pontuacao = 0
respostasAluno = []

const corretas = {
p1:"2",
p2:"2",
p3:"2",
p4:"1",
p5:"2",
p6:"2",
p7:"2"
}

for(let i=1;i<=7;i++){

let resposta = document.querySelector('input[name="p'+i+'"]:checked')

if(resposta){

respostasAluno.push(resposta.value)

if(resposta.value == corretas["p"+i]){
pontuacao++
}

}else{

respostasAluno.push("NA")

}

}

let mensagem=""

if(pontuacao >=6){
mensagem="Excelente ética profissional!"
}
else if(pontuacao>=4){
mensagem="Bom, mas ainda pode melhorar."
}
else{
mensagem="Reflita mais sobre ética na programação."
}

document.getElementById("resultado").innerHTML =
"Pontuação: "+pontuacao+" / 7<br>"+mensagem

document.getElementById("baixar").style.display="inline"

}

function baixarCSV(){

let nome = document.getElementById("nome").value

let csv = "Nome,P1,P2,P3,P4,P5,P6,P7,Pontuacao\n"

csv += nome + "," + respostasAluno.join(",") + "," + pontuacao

let blob = new Blob([csv], { type: "text/csv" })

let link = document.createElement("a")

link.href = URL.createObjectURL(blob)

link.download = "resultado_etica_programacao.csv"

link.click()

}
