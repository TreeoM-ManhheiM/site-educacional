# ==========================================
# IMPORTAR BIBLIOTECA
# ==========================================

import numpy as np

# ==========================================
# CRIAÇÃO DE ARRAYS
# ==========================================

a = np.array([1,2,3,4])
b = np.zeros(4)
c = np.ones(4)
d = np.empty(4)

print(np.arange(0,10,2))
print(np.linspace(0,10,5))
print(np.logspace(1,3,3))

print(np.eye(3))
print(np.identity(3))

# ==========================================
# INFORMAÇÕES SOBRE ARRAY
# ==========================================

print(a.shape)
print(a.size)
print(a.ndim)
print(a.dtype)
print(a.itemsize)

# ==========================================
# OPERAÇÕES MATEMÁTICAS
# ==========================================

print(np.add(a,c))
print(np.subtract(a,c))
print(np.multiply(a,c))
print(np.divide(a,c))

print(np.power(a,2))
print(np.sqrt(a))
print(np.abs([-1,-2,-3]))
print(np.exp(a))
print(np.log(a))
print(np.log10(a))

# ==========================================
# ESTATÍSTICA
# ==========================================

print(np.mean(a))
print(np.median(a))
print(np.std(a))
print(np.var(a))

print(np.sum(a))
print(np.min(a))
print(np.max(a))

print(np.argmin(a))
print(np.argmax(a))

print(np.percentile(a,50))

# ==========================================
# MANIPULAÇÃO DE ARRAYS
# ==========================================

print(np.reshape(a,(2,2)))
print(np.resize(a,(2,3)))

print(np.append(a,5))
print(np.insert(a,1,10))
print(np.delete(a,2))

print(np.concatenate((a,c)))
print(np.split(a,2))

print(np.vstack((a,c)))
print(np.hstack((a,c)))

# ==========================================
# ORDENAÇÃO E BUSCA
# ==========================================

print(np.sort(a))
print(np.argsort(a))

print(np.where(a>2))
print(np.searchsorted(a,3))

# ==========================================
# ÁLGEBRA LINEAR
# ==========================================

A = np.array([[1,2],[3,4]])
B = np.array([[5,6],[7,8]])

print(np.dot(A,B))
print(np.matmul(A,B))

print(np.linalg.inv(A))
print(np.linalg.det(A))

print(np.linalg.eig(A))

b2 = np.array([5,11])
print(np.linalg.solve(A,b2))

# ==========================================
# NÚMEROS ALEATÓRIOS
# ==========================================

print(np.random.rand(3))
print(np.random.randn(3))
print(np.random.randint(1,10,3))

print(np.random.choice(a))
np.random.shuffle(a)
print(a)

# ==========================================
# FUNÇÕES TRIGONOMÉTRICAS
# ==========================================

print(np.sin(a))
print(np.cos(a))
print(np.tan(a))

print(np.arcsin(np.sin(1)))
print(np.arccos(np.cos(1)))
print(np.arctan(np.tan(1)))
