function preJogo(i) { const peca = maoAtual[i]; // Se a mesa estiver vazia, joga direto if (mesaAtual.length === 0) { socket.emit('jogarPeca', { index: i, lado: 'dir' }); return; } const pEsq = mesaAtual[0][0]; const pDir = mesaAtual[mesaAtual.length - 1][1]; const serveEsq = (peca[0] === pEsq || peca[1] === pEsq); const serveDir = (peca[0] === pDir || peca[1] === pDir); if (serveEsq && serveDir) { // Se servir nos dois, para não bugar com modal, joga na direita por padrão // Ou você pode descomentar a linha do modal abaixo se ele estiver funcionando // document.getElementById('modalLado').style.display = 'flex'; idxSel = i; socket.emit('jogarPeca', { index: i, lado: 'dir' }); } else if (serveEsq) { socket.emit('jogarPeca', { index: i, lado: 'esq' }); } else if (serveDir) { socket.emit('jogarPeca', { index: i, lado: 'dir' }); } else { alert("Essa peça não encaixa em lugar nenhum!"); } }
